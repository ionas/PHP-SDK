This class is a default Implementation for Log4php

it is used to Log Messages to files and is injected by default

$config = array(
'filename' => 'payone/exception.log',
'max_file_size' => '500KB',
'max_file_count' => 10,
);
$logger = new Payone_Protocol_Logger_Log4php();
$logger->log('MESSAGE');
